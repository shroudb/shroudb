ShrouDB has been deployed as a StatefulSet.

Get the RESP3 endpoint:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "shroudb.fullname" . }} {{ .Values.service.resp3Port }}:{{ .Values.service.resp3Port }}
{{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc {{ include "shroudb.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "RESP3: $SERVICE_IP:{{ .Values.service.resp3Port }}"
{{- end }}

  kubectl port-forward svc/{{ include "shroudb.fullname" . }} {{ .Values.service.metricsPort }}:{{ .Values.service.metricsPort }}
